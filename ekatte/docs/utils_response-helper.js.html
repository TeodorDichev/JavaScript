/**
 * Sends a standardized JSON response to the client.
 * This utility handles both successful and error responses, logging errors 
 * to the console for debugging when the status code is 400 or higher.
 *
 * @param {http.ServerResponse} res - The Node.js HTTP response object.
 * @param {number} status - The HTTP status code (e.g., 200, 400, 500).
 * @param {Object} [options={}] - The response payload options.
 * @param {string} [options.message=""] - A human-readable message describing the result.
 * @param {Array<string>|string} [options.errors=[]] - A list of error details or a single error message.
 * @param {Object|Array|null} [options.data=null] - The actual data payload to be returned.
 * @returns {void}
 * * @example
 * * // Success Response
 * sendResponse(res, 200, { data: { id: 1 }, message: "Success" });
 * * // Error Response
 * sendResponse(res, 404, { message: "Not Found", errors: ["Item does not exist"] });
 */
export function sendResponse(res, status, { message = "", errors = [], data = null } = {}) {
    const successful = status < 400;

    if (!successful) {
        console.error(`[API Error ${status}]: ${message}`);
        if (errors.length > 0) console.error("Details:", errors);
    }

    res.writeHead(status, { "Content-Type": "application/json" });
    
    res.end(JSON.stringify({
        successful,
        message,
        errors: Array.isArray(errors) ? errors : [errors],
        data
    }));
}
